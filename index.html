Vue-use-popperjs

Installation

# With Npm
npm i vue-use-popperjs @popperjs/core@2
# With Yarn
yarn add vue-use-popperjs @popperjs/core@2
# With CDN
<script src="https://unpkg.com/@popperjs/core@2"></script>
<script src="https://unpkg.com/vue-use-popperjs/dist/umd/index.js"></script>

Basic usage

<button ref="reference">Reference</button>
<div ref="popper">Popper</div>
import { defineComponent, ref } from 'vue'
import { usePopperjs } from 'vue-use-popperjs'

defineComponent({
  setup() {
    const reference = ref()
    const popper = ref()

    usePopperjs(reference, popper, {
      placement: 'bottom'
    })

    return {
      reference,
      popper,
    }
  },
})
:global(.vue-use-popperjs-none) {
  visibility: hidden;
  pointer-events: none;
}
